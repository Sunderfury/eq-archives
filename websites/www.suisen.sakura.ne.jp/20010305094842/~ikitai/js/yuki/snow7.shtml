「雪を振らせる」の改造
画像の中だけ雪を振らせる

1.雪を振らせたい画像のタグ(赤)の両側に紫色の行を挿入する。
2.スタイルシート(青)をHEAD内へ挿入し画像の大きさに合わせ数値を変える。
3.BODYタグ内へ onload="init()" を挿入
4.JavaScript(茶)をBODY内へ挿入し太文字の所を説明に合わせて変更する。画像のタグは1番の赤色の所と同じものを入れてください。

<HTML>
<HEAD>
<TITLE>画像の中だけ雪を振らせる</TITLE>

<STYLE type="text/css">
<!-- /* 507が画像の幅、380が画像の高さ */
.snowArea { width:507; height:380; clip:rect(0px 507px 380px 0px); }
#snowlayerNN { position:relative; }
#snowlayerIE { position:absolute; top:0; left:0; }
-->
</STYLE>

</HEAD>

<BODY bgcolor="#ffffff" onload="init()">


<center>
<TABLE border="0" cellpadding="0" cellspacing="0"><TR><TD><DIV id="snowlayerNN" class="snowArea">
<IMG src="originnightyew.jpg" width="507" height="380" border="0">
<DIV id="snowlayerIE" class="snowArea"></DIV></DIV></TD></TR></TABLE>
</center>


<SCRIPT LANGUAGE="JavaScript1.2">
<!-- Begin
//<!-- Original:  Altan (snow@altan.hr) -->
//<!-- Web Site:  http://www.altan.hr/snow -->
// Script arranged by Hisa (http://www.suisen.sakura.ne.jp/~ikitai/js/)

var no = 15; // １画面に出る雪画像の枚数
var speed = 10; // smaller number moves the snow faster
var snowflake = "snow.gif"; // 雪の画像
var imgW = 507; // 画像の幅
var imgH = 380; // 画像の高さ
var imgTag= '<IMG src="originnightyew.jpg" width="507" height="380" border="0">';    // 画像のタグ

var ns4up = (document.layers) ? 1 : 0;  // browser sniffer
var ie4up = (document.all) ? 1 : 0;
var dx, xp, yp;    // coordinate and position variables
var am, stx, sty;  // amplitude and step variables
var i;

dx = new Array();
xp = new Array();
yp = new Array();
am = new Array();
stx = new Array();
sty = new Array();

function init() {
    if (ns4up) {
        lay = document.layers["snowlayerNN"];
        lay.document.open();
        lay.document.write('<layer name="imglay">'+imgTag+'</layer>');
    } else if (ie4up) {
        lay = document.all("snowlayerIE");
    }

    for (i = 0; i < no; ++ i) {
        dx[i] = 0;                        // set coordinate variables
        xp[i] = Math.random()*(imgW-50);  // set position variables
        yp[i] = Math.random()*imgH;
        am[i] = Math.random()*20;         // set amplitude variables
        stx[i] = 0.02 + Math.random()/10; // set step variables
        sty[i] = 0.7 + Math.random();     // set step variables
        if (ns4up) {                      // set layers
            if (i == 0) {
                lay.document.write("<layer name=\"dot"+ i +"\" left=\"15\" top=\"15\" visibility=\"show\"><img src=\""
                + snowflake + "\" border=\"0\"></layer>");
            } else {
                lay.document.write("<layer name=\"dot"+ i +"\" left=\"15\" top=\"15\" visibility=\"show\"><img src=\""
                + snowflake + "\" border=\"0\"></layer>");
            }
        } else if (ie4up) {
            if (i == 0) {
                lay.insertAdjacentHTML('BeforeEnd',"<div id=\"dot"+ i +"\" style=\"POSITION: absolute; Z-INDEX: "
                + i +"; VISIBILITY: visible; TOP: 15px; LEFT: 15px; width:1;\"><img src=\"" + snowflake
                + "\" border=\"0\"></div>");
            } else {
                lay.insertAdjacentHTML('BeforeEnd',"<div id=\"dot"+ i +"\" style=\"POSITION: absolute; Z-INDEX: "
                + i +"; VISIBILITY: visible; TOP: 15px; LEFT: 15px; width:1;\"><img src=\"" + snowflake
                + "\" border=\"0\"></div>");
            }
        }
    }

    if (ns4up) {
        lay.document.close();
        lay.document.layers["imglay"].moveTo(0,0);
        snowNS();
    } else if (ie4up) {
        snowIE();
    }
}

function snowNS() {  // Netscape main animation function
    for (i = 0; i < no; ++ i) {  // iterate for every dot
        yp[i] += sty[i];
        if (yp[i] > imgH) {
            xp[i] = Math.random()*(imgW-30);
            yp[i] = 0;
            stx[i] = 0.02 + Math.random()/10;
            sty[i] = 0.7 + Math.random();
            doc_width = self.innerWidth;
            doc_height = self.innerHeight;
        }
        dx[i] += stx[i];
        lay.document.layers["dot"+i].top = yp[i];
        lay.document.layers["dot"+i].left = xp[i] + am[i]*Math.sin(dx[i]);
    }
    setTimeout("snowNS()", speed);
}

function snowIE() {  // IE main animation function
    for (i = 0; i < no; ++ i) {  // iterate for every dot
        yp[i] += sty[i];
        if (yp[i] > imgH) {
            xp[i] = Math.random()*(imgW-30);
            yp[i] = 0;
            stx[i] = 0.02 + Math.random()/10;
            sty[i] = 0.7 + Math.random();
            doc_width = document.body.clientWidth;
            doc_height = document.body.clientHeight;
        }
        dx[i] += stx[i];
        lay.document.all["dot"+i].style.pixelTop = yp[i];
        lay.document.all["dot"+i].style.pixelLeft = xp[i] + am[i]*Math.sin(dx[i]);
    }
    setTimeout("snowIE()", speed);
}

// End -->
</SCRIPT>

</BODY>
</HTML>
